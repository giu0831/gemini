        setTimeout(() => {

    function initNamingGrid() {
        const grid = document.getElementById('naming-grid'); grid.innerHTML = '';
        namingChars.forEach((char, index) => {
            let div = document.createElement('div'); div.className = 'naming-char'; div.innerText = char; div.id = 'char-' + index;
            grid.appendChild(div);
        }); updateNamingUI();
    }

    function updateNamingUI() {
        let displayStr = playerName.split('').join(' ');
        if (playerName.length < 8) displayStr += (playerName.length > 0 ? ' ' : '') + '_ '.repeat(8 - playerName.length).trim();
        document.getElementById('player-name-display').innerText = displayStr;
        for(let i = 0; i < namingChars.length; i++) {
            let el = document.getElementById('char-' + i);
            if (i === namingIndex) el.classList.add('active'); else el.classList.remove('active');
        }
    }

    window.addEventListener('keydown', (e) => { keys[e.key] = true; processInput(e.key); });
    window.addEventListener('keyup', (e) => { keys[e.key] = false; });

    function processInput(key) {
        if (gameState === 'NAMING') {
            if (key === 'ArrowLeft') { if (namingIndex % 7 !== 0) namingIndex--; }
            else if (key === 'ArrowRight') { if (namingIndex % 7 !== 6 && namingIndex < 27) namingIndex++; }
            else if (key === 'ArrowUp') { if (namingIndex >= 7) namingIndex -= 7; }
            else if (key === 'ArrowDown') { if (namingIndex + 7 <= 27) namingIndex += 7; }
            else if (key === 'z' || key === 'Z' || key === 'Enter') {
                let char = namingChars[namingIndex];
                if (char === '<-') playerName = playerName.slice(0, -1);
                else if (char === 'OK') { if (playerName.trim().length > 0) checkNameEasterEggs(); }
                else if (playerName.length < 8) playerName += char;
            } else if (key === 'x' || key === 'X' || key === 'Backspace') playerName = playerName.slice(0, -1);
            updateNamingUI();
        } 
        else if (gameState === 'CONFIRM_NAME') {
            if (key === 'ArrowLeft' || key === 'ArrowRight') { confirmSelection = (confirmSelection === 0) ? 1 : 0; updateConfirmCursor(); }
            if (key === 'z' || key === 'Z' || key === 'Enter') { 
                if (blockNameChoice || confirmSelection === 1) backToNaming(); 
                else confirmName(); 
            } 
        } 
        else if (gameState === 'SAVE_SCREEN') { if (key === 'z' || key === 'Z' || key === 'Enter') startGame(); } 
        else if (gameState === 'MAIN_MENU') {
            if (key === 'ArrowLeft') { selectedButton = (selectedButton - 1 + 4) % 4; updateButtons(); }
            else if (key === 'ArrowRight') { selectedButton = (selectedButton + 1) % 4; updateButtons(); }
            else if (key === 'z' || key === 'Z' || key === 'Enter') selectAction();
        } 
        else if (gameState === 'MENU_SELECTION') {
            if (key === 'ArrowUp') { menuSelectionIndex = Math.max(0, menuSelectionIndex - 1); updateMenuSelectionUI(); }
            else if (key === 'ArrowDown') { menuSelectionIndex = Math.min(menuOptions.length - 1, menuSelectionIndex + 1); updateMenuSelectionUI(); }
            else if (key === 'z' || key === 'Z' || key === 'Enter') executeMenuAction();
            else if (key === 'x' || key === 'X') showMainMenu();
        } 
        else if (gameState === 'TARGETING') { if (key === 'z' || key === 'Z' || key === 'Enter') stopTarget(); } 
        else if (gameState === 'BOSS_DIALOGUE') { if (key === 'z' || key === 'Z' || key === 'Enter') startDefendPhase(); }
    }

        document.getElementById('confirm-screen').style.display = 'none';
        document.getElementById('save-screen').style.display = 'flex';
        gameState = 'SAVE_SCREEN';
    }

        document.getElementById('save-screen').style.display = 'none';
        document.getElementById('game-container').style.display = 'flex';
        showMainMenu();
    }

        gameState = 'MAIN_MENU'; textArea.style.display = 'block';

        updateButtons();
    }

    let selectedButton = 0; let menuSelectionIndex = 0; let menuOptions = [];

    function selectAction() {
        if (selectedButton === 0) startTargeting();
        else if (selectedButton === 1) {
            gameState = 'MENU_SELECTION'; menuOptions = ["* Controlla", "* Chiedi un 6", "* Recita Dante"]; menuSelectionIndex = 0; updateMenuSelectionUI();
        } else if (selectedButton === 2) {
            gameState = 'MENU_SELECTION'; menuOptions = ["* Gomma (Cura 50 HP)"]; menuSelectionIndex = 0; updateMenuSelectionUI();
        } else if (selectedButton === 3) {
            textArea.innerHTML = "* La campanella non è ancora suonata. Non scapperai."; setTimeout(bossTurn, 2000);
        }
    }

    function updateMenuSelectionUI() {
        let html = "";
        menuOptions.forEach((opt, i) => {
            let cursor = (i === menuSelectionIndex) ? "<span style='color:red; margin-right:10px;'>❤</span>" : "<span style='margin-right:22px;'></span>";
            html += `<div>${cursor}${opt}</div>`;
        });
        textArea.innerHTML = html;
    }

    function executeMenuAction() {
        const selectedOpt = menuOptions[menuSelectionIndex];
        if (selectedOpt.includes("Controlla")) textArea.innerHTML = "* LA PROF. AISA 99 ATK 99 DEF.<br>* Distorce lo spazio e il tempo scolastico.";

        else if (selectedOpt.includes("Dante")) { textArea.innerHTML = "* Reciti l'Inferno.<br>* Si commuove. Attacchi rallentati."; danteBuff = true; }
        else if (selectedOpt.includes("Gomma")) { playerHP = Math.min(MAX_HP, playerHP + 50); updateHP(); textArea.innerHTML = "* Hai cancellato i tuoi errori!"; }
        setTimeout(bossTurn, 2500);
    }

    function startTargeting() {
        gameState = 'TARGETING'; textArea.style.display = 'none'; targetBar.style.display = 'block';
        targetX = 0; targetDirection = 1; targetLoop();

    function targetLoop() {
        if (gameState !== 'TARGETING') return; targetX += 8 * targetDirection;
        const limit = (battleBox.offsetWidth || 540) - 10;
        if (targetX > limit) { targetX = limit; targetDirection = -1; } else if (targetX < 0) { targetX = 0; targetDirection = 1; }
        targetCursor.style.left = targetX + 'px'; targetReqId = requestAnimationFrame(targetLoop);
    }

            dmgText.innerText = "MISS"; dmgText.classList.add('miss-text'); dmgText.style.display = 'block';
            bossImg.classList.add('dodging'); bossPlac.classList.add('dodging');
            setTimeout(() => { dmgText.style.display = 'none'; dmgText.classList.remove('miss-text'); bossImg.classList.remove('dodging'); bossPlac.classList.remove('dodging'); bossTurn(); }, 1000);
        }, 400);
    }

    function bossTurn() {
        gameState = 'BOSS_DIALOGUE'; targetBar.style.display = 'none'; textArea.style.display = 'block'; bossBubble.style.display = 'block';
        bossBubble.innerText = bossQuotes[currentTurn % bossQuotes.length]; textArea.innerHTML = "* (Preparati...)";
    }

    // ============================================
    // GESTIONE ARENA E ATTACCHI DINAMICI
    // ============================================
    function changeSoulColor(color) {
        if (soul.color === color) return;
        soul.color = color; soul.vy = 0; soul.grounded = false;
        battleBox.classList.remove('flash-blue', 'flash-red');
        void battleBox.offsetWidth; // trigger reflow
        battleBox.classList.add(color === 'blue' ? 'flash-blue' : 'flash-red');
    }

    function resizeBox(w, h) {
        battleBox.style.width = (window.innerWidth <= 600 ? Math.min(w, window.innerWidth - 20) : w) + 'px';
        battleBox.style.height = h + 'px';
    }

    function startDefendPhase() {
        gameState = 'DEFEND'; bossBubble.style.display = 'none'; textArea.style.display = 'none'; canvas.style.display = 'block';
        
        // Setup iniziale di default
        resizeBox(300, 300); soul.color = 'red'; soul.x = 150; soul.y = 150;
        defendFrames = 0; blasters = []; pencils = []; grades = []; chalks = []; turnEvents = [];
        invincibilityFrames = 0; currentDefendDuration = 450; 

        let cycle = currentTurn % 4;

        // PROGRAMMAZIONE EVENTI DINAMICI (SANS STYLE)
        if (cycle === 0) {
            // Turno Normale, poi schiaccia l'area
            turnEvents.push({ f: 150, action: () => resizeBox(450, 150) });
            turnEvents.push({ f: 300, action: () => resizeBox(150, 400) });
        } 
        else if (cycle === 1) {
            // Turno gravità pazza a sorpresa
            resizeBox(400, 200);
            turnEvents.push({ f: 1, action: () => changeSoulColor('blue') });
            turnEvents.push({ f: 150, action: () => changeSoulColor('red') });
            turnEvents.push({ f: 160, action: () => resizeBox(200, 400) });
            turnEvents.push({ f: 300, action: () => changeSoulColor('blue') });
        }
        else if (cycle === 2) {
            // Box piccolissimo, proiettili veloci
            resizeBox(150, 150);
            turnEvents.push({ f: 200, action: () => resizeBox(500, 200) });
            turnEvents.push({ f: 210, action: () => changeSoulColor('blue') });
        }
        else if (cycle === 3) {
            // Corridoio Lungo
            resizeBox(500, 100);
            turnEvents.push({ f: 180, action: () => { resizeBox(250, 250); changeSoulColor('blue'); }});
        }

        requestAnimationFrame(defendLoop);
    }

    function getSpeedModifier() { return danteBuff ? 0.7 : 1 + (currentTurn * 0.05); }

    function spawnProjectileBasedOnArea(type) {
        let cw = canvas.width; let ch = canvas.height;
        if (cw === 0 || ch === 0) return; // aspetta il resize

        if (type === 'pencil' && soul.color === 'red') {
            let py = Math.random() > 0.5 ? ch - 40 : 5;
            pencils.push({ x: cw + 20, y: py, w: 14, h: 35, dx: -3 * getSpeedModifier(), dy: 0, pointing: py > ch/2 ? 'up' : 'down' });
        }
        else if (type === 'floor_pencil' && soul.color === 'blue') {
            let ph = 20 + Math.random() * 20;
            pencils.push({ x: cw + 20, y: ch - ph - 5, w: 16, h: ph, dx: -3 * getSpeedModifier(), dy: 0, pointing: 'up' });
        }
        else if (type === 'blaster') {
            // Blaster sparano in base alle dimensioni attuali
            let isHoriz = cw > ch;
            blasters.push({
                x: isHoriz ? (Math.random()>0.5 ? 20 : cw-20) : Math.max(30, Math.min(cw-30, soul.x)),
                y: isHoriz ? Math.max(30, Math.min(ch-30, soul.y)) : (Math.random()>0.5 ? 20 : ch-20),
                dir: isHoriz ? (soul.x > cw/2 ? 'L' : 'R') : (soul.y > ch/2 ? 'U' : 'D'),
                state: 'charge', timer: 0
            });
        }
        else if (type === 'grade') {
            grades.push({ x: Math.random()>0.5?10:cw-10, y: 20+Math.random()*(ch-40), vx: (Math.random()>0.5?2:-2)*getSpeedModifier(), vy: (Math.random()>0.5?2:-2)*getSpeedModifier(), text: Math.random()>0.5?"2":"4" });
        }
    }

    function defendLoop() {
        if (gameState !== 'DEFEND') return;
        
        // AGGIORNA DIMENSIONI CANVAS DINAMICAMENTE
        // Questo è il cuore della meccanica "Sans": il canvas segue le dimensioni CSS in tempo reale
        if (canvas.width !== battleBox.clientWidth || canvas.height !== battleBox.clientHeight) {
            canvas.width = battleBox.clientWidth;
            canvas.height = battleBox.clientHeight;
        }

        ctx.clearRect(0, 0, canvas.width, canvas.height);
        
        // ESEGUI EVENTI PROGRAMMATI
        turnEvents.forEach(ev => { if (ev.f === defendFrames) ev.action(); });

        let spd = keys['x'] ? soul.speed/2 : soul.speed;

        // FISICA ANIMA
        if (soul.color === 'red') {
            let dx = 0; let dy = 0;
            if (keys['ArrowLeft'] || keys['a']) dx -= 1; if (keys['ArrowRight'] || keys['d']) dx += 1;
            if (keys['ArrowUp'] || keys['w']) dy -= 1; if (keys['ArrowDown'] || keys['s']) dy += 1;
            if (dx !== 0 && dy !== 0) { let l = Math.sqrt(dx*dx + dy*dy); dx /= l; dy /= l; }
            soul.x += dx * spd; soul.y += dy * spd;
        } else if (soul.color === 'blue') {
            let dx = 0;
            if (keys['ArrowLeft'] || keys['a']) dx -= 1; if (keys['ArrowRight'] || keys['d']) dx += 1;
            soul.x += dx * spd;
            soul.vy += soul.gravity;
            if ((keys['ArrowUp'] || keys['w']) && soul.grounded) { soul.vy = soul.jump; soul.grounded = false; }
            if (!(keys['ArrowUp'] || keys['w']) && soul.vy < -2.0) soul.vy = -2.0;
            soul.y += soul.vy;
        }

        // CONFINI DINAMICI (L'anima non può uscire dal riquadro anche se si restringe)
        if (soul.x < 15) soul.x = 15;
        if (soul.x > canvas.width - 15) soul.x = canvas.width - 15;
        if (soul.y < 15) { soul.y = 15; if(soul.color==='blue') soul.vy = 0; }
        
        if (soul.y >= canvas.height - 15) { 
            soul.y = canvas.height - 15; 
            if(soul.color==='blue') { soul.vy = 0; soul.grounded = true; } 
        } else { if(soul.color==='blue') soul.grounded = false; }

        if (invincibilityFrames > 0) invincibilityFrames--;

        // DISEGNO ANIMA
        let sx = soul.x; let sy = soul.y; let sr = 4;
        if (invincibilityFrames === 0 || Math.floor(invincibilityFrames / 6) % 2 === 0) {
            ctx.fillStyle = (soul.color === 'blue') ? '#0066ff' : 'red'; 
            ctx.font = '24px Arial'; ctx.textAlign = 'center'; ctx.fillText('❤', soul.x, soul.y + 8); 
        }

        // SPAWN ATTACCHI DINAMICI BASATI SULLA FASE ATTUALE
        let cycle = currentTurn % 4;
        if (defendFrames < currentDefendDuration - 60) {
            if (cycle === 0) { if (defendFrames % 40 === 0) spawnProjectileBasedOnArea('pencil'); if (defendFrames % 90 === 0) spawnProjectileBasedOnArea('blaster'); }
            else if (cycle === 1) { if (defendFrames % 50 === 0) spawnProjectileBasedOnArea(soul.color === 'blue' ? 'floor_pencil' : 'pencil'); }
            else if (cycle === 2) { if (defendFrames % 40 === 0) spawnProjectileBasedOnArea('grade'); }
            else if (cycle === 3) { if (defendFrames % 60 === 0) spawnProjectileBasedOnArea(soul.color === 'blue' ? 'floor_pencil' : 'blaster'); }
        }

        // UPDATE PROIETTILI
        for (let i = pencils.length - 1; i >= 0; i--) {
            let p = pencils[i]; p.x += p.dx; p.y += p.dy;
            ctx.fillStyle = 'white'; ctx.fillRect(p.x, p.y, p.w, p.h); ctx.beginPath();
            if (p.pointing === 'up') { ctx.moveTo(p.x, p.y); ctx.lineTo(p.x + p.w / 2, p.y - 15); ctx.lineTo(p.x + p.w, p.y); ctx.fill(); ctx.fillRect(p.x, p.y + p.h + 2, p.w, 6); } 
            else { ctx.moveTo(p.x, p.y + p.h); ctx.lineTo(p.x + p.w / 2, p.y + p.h + 15); ctx.lineTo(p.x + p.w, p.y + p.h); ctx.fill(); ctx.fillRect(p.x, p.y - 8, p.w, 6); }

            let hitX = (sx + sr > p.x) && (sx - sr < p.x + p.w);
            let hitY = (p.pointing === 'up') ? (sy + sr > p.y - 15) && (sy - sr < p.y + p.h + 10) : (sy + sr > p.y - 10) && (sy - sr < p.y + p.h + 15);
            if (hitX && hitY) takeDamage();
            if (p.x < -50 || p.x > canvas.width+50 || p.y < -50 || p.y > canvas.height+50) pencils.splice(i, 1);
        }

        for (let i = blasters.length - 1; i >= 0; i--) {
            let b = blasters[i]; b.timer++;
            ctx.save(); ctx.translate(b.x, b.y);
            if (b.state === 'charge') ctx.translate(Math.random()*4-2, Math.random()*4-2);
            let isFlashing = (b.state === 'charge' && b.timer > 35 && b.timer % 4 < 2);
            ctx.fillStyle = isFlashing ? 'black' : 'white'; ctx.fillRect(-15, -10, 30, 20); 
            ctx.fillStyle = isFlashing ? 'white' : 'black'; ctx.fillRect(-2, -8, 4, 16); 
            ctx.strokeStyle = isFlashing ? 'black' : 'white'; ctx.lineWidth = 2; ctx.beginPath();
            if(b.dir === 'R') { ctx.moveTo(10, -10); ctx.lineTo(20, -15); ctx.moveTo(10, 10); ctx.lineTo(20, 15); }
            else if(b.dir === 'L') { ctx.moveTo(-10, -10); ctx.lineTo(-20, -15); ctx.moveTo(-10, 10); ctx.lineTo(-20, 15); }
            else if(b.dir === 'D') { ctx.moveTo(-10, 10); ctx.lineTo(-15, 20); ctx.moveTo(10, 10); ctx.lineTo(15, 20); }
            else if(b.dir === 'U') { ctx.moveTo(-10, -10); ctx.lineTo(-15, -20); ctx.moveTo(10, -10); ctx.lineTo(15, -20); }
            ctx.stroke();

            if (b.timer > 50 && b.state === 'charge') { b.state = 'fire'; b.timer = 0; } 
            ctx.restore();

            if (b.state === 'fire') {
                let beamWidth = b.timer < 8 ? b.timer * 4 : (35 - (b.timer - 8) * 1.5);
                if (beamWidth > 0) {
                    ctx.fillStyle = 'white'; let beam = {l:0, r:0, t:0, b:0};
                    if (b.dir === 'R') { ctx.fillRect(b.x + 16, b.y - beamWidth/2, 1000, beamWidth); beam={l:b.x+16, r:1000, t:b.y-beamWidth/2, b:b.y+beamWidth/2}; }
                    else if (b.dir === 'L') { ctx.fillRect(-100, b.y - beamWidth/2, b.x + 84, beamWidth); beam={l:-100, r:b.x-16, t:b.y-beamWidth/2, b:b.y+beamWidth/2}; }
                    else if (b.dir === 'D') { ctx.fillRect(b.x - beamWidth/2, b.y + 16, beamWidth, 1000); beam={l:b.x-beamWidth/2, r:b.x+beamWidth/2, t:b.y+16, b:1000}; }
                    else { ctx.fillRect(b.x - beamWidth/2, -100, beamWidth, b.y + 84); beam={l:b.x-beamWidth/2, r:b.x+beamWidth/2, t:-100, b:b.y-16}; }
                    
                    if (b.timer > 4 && b.timer < 25) { if (sx + sr > beam.l + 2 && sx - sr < beam.r - 2 && sy + sr > beam.t + 2 && sy - sr < beam.b - 2) takeDamage(); }
                }
                if (b.timer > 35) blasters.splice(i, 1);
            }
        }

        for (let i = grades.length - 1; i >= 0; i--) {
            let g = grades[i]; g.x += g.vx; g.y += g.vy;
            if (g.x <= 15 || g.x >= canvas.width - 15) g.vx *= -1;
            if (g.y <= 15 || g.y >= canvas.height - 15) g.vy *= -1;
            ctx.fillStyle = 'red'; ctx.font = 'bold 26px VT323'; ctx.textAlign = 'center'; ctx.textBaseline = 'middle'; ctx.fillText(g.text, g.x, g.y);
            if (Math.hypot(sx - g.x, sy - g.y) < sr + 10) takeDamage();
            if (defendFrames > currentDefendDuration - 30) grades.splice(i, 1);
        }

        defendFrames++;
        if (defendFrames > currentDefendDuration) { currentTurn++; danteBuff = false; showMainMenu(); } 
        else { animationId = requestAnimationFrame(defendLoop); }
    }

    function takeDamage() {
        if (invincibilityFrames > 0) return; playerHP -= 8; if (playerHP < 0) playerHP = 0; updateHP(); invincibilityFrames = 80;
        if (playerHP === 0 && gameState !== 'GAME_OVER') {
            gameState = 'GAME_OVER'; cancelAnimationFrame(animationId);
            textArea.style.display = 'block'; canvas.style.display = 'none';
            battleBox.style.width = window.innerWidth <= 600 ? '95%' : '540px'; battleBox.style.height = '160px';
            textArea.innerHTML = "* Il tuo percorso scolastico finisce qui.<br>* Game Over.";
        }
    }

    const mobileKeyMap = {'m-up':'ArrowUp','m-down':'ArrowDown','m-left':'ArrowLeft','m-right':'ArrowRight','m-z':'z','m-x':'x'};
    for (const [id, key] of Object.entries(mobileKeyMap)) {
        const btn = document.getElementById(id);
        btn.addEventListener('touchstart', (e) => { e.preventDefault(); keys[key] = true; processInput(key); });
        btn.addEventListener('touchend', (e) => { e.preventDefault(); keys[key] = false; });
    }
